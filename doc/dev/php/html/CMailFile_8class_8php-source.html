00194 // Splits a string by RFC2045 semantics (76 chars per line, end with \r\n).
00195 // This is not in all PHP versions so I define one here manuall.

00197 function my_chunk_split($str)
00198 {
00199   $stmp = $str;
00200   $len = strlen($stmp);
00201   $out = "";
00202   while ($len > 0) {
00203     if ($len >= 76) {
00204       $out = $out . substr($stmp, 0, 76) . "\r\n";
00205       $stmp = substr($stmp, 76);
00206       $len = $len - 76;
00207     }
00208     else {
00209       $out = $out . $stmp . "\r\n";
00210       $stmp = ""; $len = 0;
00211     }
00212   }
00213   return $out;
